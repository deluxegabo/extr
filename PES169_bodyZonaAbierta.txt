			$(document).ready(function() {
				$("#fec").datepicker({
					changeMonth : true,
					changeYear : true,
					dateFormat : "dd/mm/yy",
					minDate : 1

				}).val(function() {
					var d = new Date();
					var dia = d.getDate() + 1 + '';
					var mes = d.getMonth() + 1 + '';
					var anio = d.getFullYear();

					var mesConc = (mes.length === 1 ? mes = '0' + mes : mes);
					var diaConc = (dia.length === 1 ? dia = '0' + dia : dia);
					var fech = diaConc + "/" + mesConc + "/" + anio;

					return fech;
				});
				$("#fec").attr("readonly", true);
			});
		</script>
